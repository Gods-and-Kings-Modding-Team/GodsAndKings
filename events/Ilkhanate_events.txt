#Safaviyya Revolt
country_event = {
	id = ilkhanate_flavor.1
	title = ilkhanate_flavor.1.t
	desc = ilkhanate_flavor.1.d
	picture = DHIMMI_ESTATE_UPSET_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		OR = {
			NOT = { exists = ARL }
			is_subject_of = ILK
		}
		num_of_owned_provinces_with = {
			value = 4
			OR = {
				area = azerbaijan_area
				area = shirvan_area
			}
			controlled_by = REB
		}
		has_country_flag = control_lost_ilk
	}
	
	immediate = {
		hidden_effect = {
			2212 = {
				add_core = ARL
			}
			5324 = {
				add_core = ARL
			}
			4338 = {
				add_core = ARL
			}
			414 = {
				add_core = ARL
			}
			4300 = {
				add_core = ARL
			}
			424 = {
				add_core = ARL
			}
			if = {
				limit = {
					ARL = {
						is_subject = yes
					}
				}
				grant_independence = ARL
			}
			else = {
				release = ARL
			}
			ARL = {
				capital_scope = {
						build_to_forcelimit = {
						infantry = 0.7
						cavalry = 0.3
					}
				}
				add_government_reform = feudal_theocracy
			}
			change_variable = {
				which = Ilkhanate_Vassal_Count
				value = 1
			}
		}
	}
	
	option = {
		name = ilkhanate_flavor.1.a
		add_prestige = -25
		
	}
	option = {
		name = ilkhanate_flavor.1.b
		switch_tag = ARL
	}
}
