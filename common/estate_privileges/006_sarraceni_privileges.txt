estate_sarraceni_guaranteed_autonomy = {
	icon = privilege_grant_autonomy
	max_absolutism = -20
	conditional_modifier = { trigger = { has_absolutism_reduction_for_estate_privileges = yes } modifier = { max_absolutism = 1 } }
	loyalty = 0.1
	influence = 0.1

	is_valid = {
		any_owned_province = {
			NOT = { religion_group = root }
		}
	}

	on_granted = {
		custom_tooltip = estate_sarraceni_guaranteed_autonomy_effect_tooltip
		hidden_effect = {
			set_country_flag = sarraceni_tolerates_wrong_religion
		}
	}
	on_granted_province = {
		if = {
			limit = {
				NOT = { religion_group = root }
			}
			add_province_modifier = {
				name = estate_sarraceni_guaranteed_autonomy_mod
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = estate_sarraceni_guaranteed_autonomy_mod
			}
			remove_province_modifier = estate_sarraceni_guaranteed_autonomy_mod
		}
	}

	on_revoked = {
		custom_tooltip = estate_dhimmi_guaranteed_autonomy_revoke_effect_tooltip
		hidden_effect = {
			clr_country_flag = sarraceni_tolerates_wrong_religion
		}
	}
	on_revoked_province = {
		if = {
			limit = {
				has_province_modifier = estate_sarraceni_guaranteed_autonomy_mod
			}
			remove_province_modifier = estate_sarraceni_guaranteed_autonomy_mod
		}
	}

	on_invalid = {
		custom_tooltip = estate_dhimmi_guaranteed_autonomy_revoke_effect_tooltip
		hidden_effect = {
			clr_country_flag = sarraceni_tolerates_wrong_religion
		}
	}
	on_invalid_province = {
		if = {
			limit = {
				has_province_modifier = estate_sarraceni_guaranteed_autonomy_mod
			}
			remove_province_modifier = estate_sarraceni_guaranteed_autonomy_mod
		}
	}

	penalties = { }
	benefits = { }

	ai_will_do = {
		factor = 1000
		modifier = {
			factor = 0
			NOT = {
				any_owned_province = {
					NOT = { religion_group = ROOT }
				}
			}
		}
		modifier = {
			factor = 0
			is_subject = yes
			overlord = {
				religion = ROOT
				has_idea_group = religious_ideas
			}
		}
	}
}
