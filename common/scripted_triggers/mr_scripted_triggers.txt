has_unified_culture = {
	custom_trigger_tooltip = {
	tooltip = has_unified_culture_TT
		all_province = {
			OR = {
				NOT = { culture = PREV }
				AND = {
					country_or_non_sovereign_subject_holds = PREV
					OR = {
						owned_by = PREV
						owner = {
							is_colonial_nation = yes
						}
					}
				}
			}
		}
	}
}

has_not_unlocked_t1_elite_units = {

	NOT = { has_country_modifier = elite_sapper_1 }
	NOT = { has_country_modifier = elite_vanguard_1 }
