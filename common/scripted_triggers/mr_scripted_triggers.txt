has_unified_culture = {
	custom_trigger_tooltip = {
	tooltip = has_unified_culture_TT
		all_province = {
			OR = {
				NOT = { culture = PREV }
				AND = {
					country_or_non_sovereign_subject_holds = PREV
					OR = {
						owned_by = PREV
						owner = {
							is_colonial_nation = yes
						}
					}
				}
			}
		}
	}
}

# Used for repeated mission in crusader states missions
cs_call_for_crusade_trigger = {
	OR = {
		AND = {
			# Popeman likes us or has to listen to us
			OR = {
				PAP = {
					has_opinion = {
						who = from
						value = 150
					}
				}
				papal_influence = 100
			}
			# A great European Prince must sponsor the Crusade
			any_known_country = {
				government = monarchy
				OR = {
					government_rank = 2
					total_development = 100
				}
				religion = from
				OR = {
					has_opinion = {
						who = from
						value = 150
					}
					owes_favors = {
						who = from
						value = 30
					}
				}
			}
		}
		# Already own Jerusalem
		owns = 379
	}
}

eng_opener_missions_progress_trigger = {
	if = {
		limit = {
			check_variable = {
				which = engOpenerMissionProg
				value = 0
			}
			NOT = {
				check_variable = {
				which = engOpenerMissionProg
					value = 1
				}
			}
		}
		stability = 1
	}
	else_if = {
		limit = {
			check_variable = {
				which = engOpenerMissionProg
				value = 1
			}
			NOT = {
				check_variable = {
				which = engOpenerMissionProg
					value = 2
				}
			}
		}
		all_owned_province = {
			is_state_core = ROOT
		}
		army_size_percentage = 1
		navy_size_percentage = 1
	}
	else_if = {
		limit = {
			check_variable = {
				which = engOpenerMissionProg
				value = 2
			}
			NOT = {
				check_variable = {
				which = engOpenerMissionProg
					value = 3
				}
			}
		}
		NOT ={
			any_owned_province = {
				province_group = england_region
				devastation = 20
			}
		}
		grown_by_development = 100
	}
	else_if = {
		limit = {
			check_variable = {
				which = engOpenerMissionProg
				value = 3
			}
			NOT = {
				check_variable = {
				which = engOpenerMissionProg
					value = 4
				}
			}
		}
		capital_scope = {
			owned_by = ROOT
			is_prosperous = yes
			development = 30
		}
	}
	else_if = {
		limit = {
			check_variable = {
				which = engOpenerMissionProg
				value = 4
			}
		}
		calc_true_if = {
			all_owned_province = {
				OR = {
					trade_goods = grain
					trade_goods = fish
				}
				base_production = 10
			}
			amount = 5
		}
	}
}

province_has_modern_fort_trigger = {
	custom_trigger_tooltip = {
		tooltip = province_has_modern_fort_trigger_TT
		if = {
			limit = {
				ROOT = {
					NOT = { mil_tech = 14 }
				}
			}
			has_building = fort_15th
		}
		else_if = {
			limit = {
				ROOT = {
					AND = {
						mil_tech = 14
						NOT = { mil_tech = 19 }
					}
				}
			}
			has_building = fort_16th
		}
		else_if = {
			limit = {
				ROOT = {
					AND = {
						mil_tech = 19
						NOT = { mil_tech = 24 }
					}
				}
			}
			has_building = fort_17th
		}
		else_if = {
			limit = {
				ROOT = {
					mil_tech = 24
				}
			}
			has_building = fort_18th
		}
	}
}
