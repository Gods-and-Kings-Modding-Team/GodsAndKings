cb_vassalize_mission_karma = {
	valid_for_subject = no

	prerequisites = {
		karma = -33
		NOT = { karma = 33 }
		uses_karma = yes
		if = {
			limit = {
				ROOT = {
					religion = mr_maitreya
				}
			}
			FROM = {
				is_neighbor_of = ROOT
				religion_group = eastern
			}
		}
		else_if = {
			limit = {
				ROOT = {
					religion = yogavacara
				}
			}
			FROM = {
				is_neighbor_of = ROOT
				religion_group = eastern
			}
		}
		else_if = {
			limit = {
				has_global_flag =  centralized_buddh
			}
			FROM = {
				is_neighbor_of = ROOT
				religion_group = eastern
			}
		}
		else = {
			FROM = {
				is_neighbor_of = ROOT
				religion = ROOT
			}
		}
	}	

	war_goal = take_capital_vassalize_karma
}
