generic_military_missions_2 = { #naval line
	slot = 2
	generic = yes
	ai = no
	potential = { always = yes }
	
	g_mission_sea_access = {
		icon = mission_build_up_to_force_limit
		required_missions = { }
		position = 1
		trigger = {
			any_owned_province = {
				has_port = yes
				development = 15
				has_dock_building_trigger = yes
			}
		}
		effect = {
			custom_tooltip = g_mission_sea_access_tt
			tooltip = {
				random_owned_province = {
					add_unit_construction = {
						type = light_ship
						amount = 3
						speed = 1
						cost = 0
					}
					province_event = {
						id = generic_mission_events.3
					}
				}
			}
			hidden_effect = {
				random_owned_province = {
					limit = {
						has_port = yes
						development = 15
						has_dock_building_trigger = yes
					}
					add_unit_construction = {
						type = light_ship
						amount = 3
						speed = 1
						cost = 0
					}
					province_event = {
						id = generic_mission_events.3
					}
				}
			}
		}
	}
	
	g_mission_naval_projection = {
		icon = mission_build_up_to_force_limit
		required_missions = { g_mission_sea_access }
		position = 2
		trigger = {
			any_owned_province = {
				has_port = yes
				development = 25
				has_dock_building_trigger = yes
				has_shipyard_building_trigger = yes
			}
			naval_forcelimit = 30
			navy_size_percentage = 1.0
			if = {
				limit = {
					has_dlc = "Rule Britannia"
				}
				has_flagship = yes
			}
		}
		effect = {
			country_event = { #Naval Ethos
				id = generic_mission_events.4
			}
		}
	}
	
}

generic_military_missions_4 = { #military line
	slot = 4
	generic = yes
	ai = no
	potential = { always = yes }
	
	g_mission_military_ethos = {
		icon = mission_build_up_to_force_limit
		required_missions = {}
		position = 1
		trigger = {
			army_size_percentage = 1
			num_of_generals = 1
		}
		effect = {
			add_country_modifier = {
				name = mr_recruiting_effort
				duration = 3650 
			}
			#pick military modifier, after g_mission_growning_power it becomes stronger/permanent
			country_event = { #Military Ethos
				id = generic_mission_events.1
			}
		}
	}

	g_mission_expand_country = {
		icon = mission_conquer_50_development
		required_missions = { g_mission_military_ethos }
		position = 2
		trigger = {
			grown_by_development = 50
			total_development = 100
			crown_land_share = 60
		}
		effect = {
			capital_scope = {
				area = { 
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT 
				}
			}
			capital_scope = {
				region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_claim = ROOT
				}
			}
			add_country_modifier = {
				name = mr_territorial_ambitions
				duration = 9125
			}
		}
	}
	
	g_mission_growing_power = {
		icon = mission_build_up_to_force_limit
		required_missions = { g_mission_expand_country }
		position = 3
		trigger = {
			if = {
				limit = {
					OR = {
						has_dlc = "Emperor"
						has_dlc = "Rights of Man"
					}
				}
				NOT = { great_power_rank = 8 }
			}
			else = {
				total_development = 1000
			}
		}
		effect = {
			country_event = { #Military Ethos Realized
				id = generic_mission_events.2
			}
		}
	}
}
