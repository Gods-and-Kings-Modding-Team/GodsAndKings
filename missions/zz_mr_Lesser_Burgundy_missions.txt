lbg_missions_first = {
	slot = 1
	generic = no
	has_country_shield = yes
	ai = yes
	potential = {
		tag = LBG
	}

	lbg_new_prosperity = {
		icon = mission_temple_patronage
		required_missions = { }
		position = 1

		trigger = {
			num_of_owned_provinces_with = {
				value = 3
				has_building = temple
			}
		}

		effect = {
			add_country_modifier = {
				name = lbg_tax_meta_1
				duration = 10950 # 30 years
			}
		}
	}

	lbg_fill_the_valleys = {
		icon = mission_valencian_silk
		required_missions = { lbg_new_prosperity }
		position = 2

		provinces_to_highlight = {
			OR = {
				has_terrain = highlands
				has_terrain = mountain
			}
			NOT = { development = 11 }
		}

		trigger = {
			NOT = {
				any_owned_province = {
					OR = {
						has_terrain = highlands
						has_terrain = mountain
					}
					NOT = { development = 11 }
				}
			}
		}

		effect = {
			add_dip_power = 100
			add_years_of_income = 2
		}
	}

	lbg_alpine_society = {
		icon = mission_se_along_the_river
		required_missions = { lbg_fill_the_valleys lbg_initial_expansion }
		position = 4

		provinces_to_highlight = {
			OR = {
				has_terrain = highlands
				has_terrain = mountain
			}
			NOT = { prosperity = 75 }
		}

		trigger = {
			NOT = {
				any_owned_province = {
					OR = {
						has_terrain = highlands
						has_terrain = mountain
					}
					NOT = { prosperity = 75 }
				}
			}
		}

		effect = {
			unlock_estate_privilege = {
				estate_privilege = estate_burghers_mountain_expansion
			}
		}
	}
}

lbg_missions_second = {
	slot = 2
	generic = no
	has_country_shield = yes
	ai = yes
	potential = {
		tag = LBG
	}

	lbg_initial_expansion = {
		icon = mission_pf_soldier
		required_missions = { }
		position = 2

		trigger = { # Has conquered 3 provinces
			num_of_owned_provinces_with = {
				value = 9
			}
		}

		# loot them
		effect = {
			set_variable = { which = i value = 0 } # i = 0
			while = { # while (i < 3)
				limit = {
					NOT = { check_variable = { which = i value = 3 } }
				}
				change_variable = { which = i value = 1} # i++

				random_owned_province = { # Loot up to 2/2/2 dev
					limit = { nationalism = 1 } # From a newly-conquered province

					remove_base_development = { type = tax 			value = 2 }
					remove_base_development = { type = production 	value = 2 }
					remove_base_development = { type = manpower 	value = 2 }
				}
			}

			165 = {
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
			}
			1867 = {
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
			}
			1871 = {
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
			}
		}
	}
}
